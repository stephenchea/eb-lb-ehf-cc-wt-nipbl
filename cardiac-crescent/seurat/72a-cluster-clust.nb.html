Recode cluster into subclusters.

umap.3[as.vector(rownames(umap.3.1 %>% filter(cluster.subclust.flox == "3.1.1"))), "cluster.clust.flox"] <- "3.1.1"
umap.3[as.vector(rownames(umap.3.1 %>% filter(cluster.subclust.flox == "3.1.2"))), "cluster.clust.flox"] <- "3.1.2"

Generate label umap.

umap.label <- umap.3 %>% group_by(cluster.clust.flox) %>% summarize(umap.1 = median(umap.1), umap.2 = median(umap.2))

Generate cluster umap.

cluster.umap <- ggplot() +
geom_point(data = sample(umap.3), mapping = aes(x = umap.1, y = umap.2, color = cluster.clust.flox), shape = 20, stroke = 0, size = 0.5) +
geom_text(data = umap.label, mapping = aes(x = umap.1, y = umap.2, label = cluster.clust.flox), size = 2.3) +
labs(title = "Cluster 3, Nipbl FIN/+", x = "UMAP 1", y = "UMAP 2", color = "Nipbl Flox/+\nCluster") +
  guides(color = guide_legend(override.aes = list(size = 2.4))) +
theme_classic(base_size = 7) +
theme(axis.line = element_blank(), axis.ticks = element_blank(), axis.text = element_blank(), plot.title = element_text(hjust = 0.5, size = 7), legend.key.size = unit(7, "pt"))
